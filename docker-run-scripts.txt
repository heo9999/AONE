#postgres-run.sh
export PGDATA=/home/${USER}/data/postgres
docker run \
-p 5432:5432 \
--name postgres \
-e POSTGRES_PASSWORD=1q2w3e4r \
-d --restart unless-stopped \
-v $PGDATA:/var/lib/postgresql/data postgres \
postgres

#postgres using
docker exec -it postgres /bin/bash

#in postgres docker
psql -U postgres
